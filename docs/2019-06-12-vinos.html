Guardar las viñas con más de 15 vinos con reseñas.

guarda_vinas <- vinas <- vinos %>%
  select(nombre, vina) %>%
  na.omit() %>%
  group_by(vina) %>%
  summarize(n_vinos = length(unique(nombre))) %>%
  filter(n_vinos > 15)
v <- vinos %>%
  select(nombre, puntos, precio, variedad, vina) %>%
  na.omit() %>%
  semi_join(guarda_vinas, by = "vina")
